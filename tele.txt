create table
  public.paragraphs (
    paragraph_id serial,
    book_id integer null,
    paragraph_string text not null,
    wordcount integer not null,
    constraint paragraphs_pkey primary key (paragraph_id),
    constraint paragraphs_book_id_fkey foreign key (book_id) references book (book_id) on delete cascade
  ) tablespace pg_default;

create index if not exists idx_paragraphs_book_id on public.paragraphs using btree (book_id) tablespace pg_default;
CREATE INDEX idx_paragraphs_fts ON public.paragraphs USING pgroonga(paragraph_string);
